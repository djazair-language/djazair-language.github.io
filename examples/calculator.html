Example: Calculator

An interactive command line calculator script:

print("--- Djazair CLI Calculator ---")
print("Enter calculation mode or type 'exit' to quit.")

while True
    let inputStr = input("calc> ")
    if inputStr == "exit"
        break
    end
    
    # Simple evaluation split
    let parts = inputStr.split(" ")
    if parts.length() &lt; 3
        print("Invalid syntax. Format: [number] [operator] [number]")
        continue
    end
    
    let a = num(parts[0])
    let op = parts[1]
    let b = num(parts[2])
    let result = 0
    
    if op == "+"
        result = a + b
    elif op == "-"
        result = a - b
    elif op == "*"
        result = a * b
    elif op == "/"
        if b == 0
            print("Error: division by zero!")
            continue
        end
        result = a / b
    else
        print("Unknown operator: " + op)
        continue
    end
    
    print("Result: ${result}")
end

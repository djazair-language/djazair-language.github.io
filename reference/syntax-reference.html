Syntax Cheat Sheet

A dense reference sheet for developers looking for quick syntax cues:

Declarations

let varName = "value"
fn funcName(a, b = 10, ...rest)
    return a + b
end

Classes

class Parent
    init() self.x = 1 end
end
class Child is Parent
    init() super.init(); self.y = 2 end
end

Loops & Flow

if condition
    # block
elif another
    # block
else
    # block
end

for val in list
    if val == "skip" continue end
    if val == "exit" break end
    print(val)
end

Exceptions

try
    throw "error"
catch e
    print("Error: " + str(e))
end
